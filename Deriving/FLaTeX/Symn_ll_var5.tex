\resetsteps      % Reset all the commands to create a blank worksheet  

% Define the operation to be computed

\renewcommand{\operation}{ $ C \becomes A B + C where A is symmetric stored in lower triange $}

\renewcommand{\routinename}{\operation}

% Step 1a: Precondition 

\renewcommand{\precondition}{
  C = \widehat{C}
}

% Step 1b: Postcondition 

\renewcommand{\postcondition}{ 
    \mbox{ $ C \becomes A B + \widehat C $}
}

% Step 2: Invariant 
% Note: Right-hand side of equalities must be updated appropriately

\renewcommand{\invariant}{
  \FlaTwoByOne{C_T}
  {A_{BR} B_B + C_B} = 
  \FlaTwoByOne{\widehat{C}_T}
  {\widehat{C}_B}
}

% Step 3: Loop-guard 

  \renewcommand{\guard}{
    m( A_{TR} ) < m( A )
  }

% Step 4: Initialize 

\renewcommand{\partitionings}{
  $
    A \rightarrow
    \FlaTwoByTwo{A_{TL}}{*}
  {A_{BL}}{A_{BR}}
  $
    ,
    $
      B \rightarrow
      \FlaTwoByOne{B_{T}}
  {B_{B}}
  $
    ,
    $
      C \rightarrow
      \FlaTwoByOne{C_{T}}
  {C_{B}}
  $
}

\renewcommand{\partitionsizes}{
  $ A_{TR} $ is $ 0 \times 0 $,
    $ B_B $ has $ 0 $ rows,
    $ C_B $ has $ 0 $ rows
}

% Step 5a: Repartition the operands 

\renewcommand{\repartitionings}{
  $  \FlaTwoByTwo{A_{TL}}{*}
  {A_{BL}}{A_{BR}}
  \rightarrow
    \FlaThreeByThreeTL{A_{00}}{*}{*}
  {a_{10}^T}{\alpha_{11}}{*}
  {A_{20}}{a_{21}}{A_{22}}
  $,
    $  \FlaTwoByOne{ B_T }
  { B_B }
  \rightarrow
    \FlaThreeByOneT{B_0}
  {b_1^T}
  {B_2}
  $
    ,
    $  \FlaTwoByOne{ C_T }
  { C_B }
  \rightarrow
    \FlaThreeByOneT{C_0}
  {c_1^T}
  {C_2}
  $
}

\renewcommand{\repartitionsizes}{
  $ \alpha_{11} $ is $ 1 \times 1 $,
    $ b_1 $ has $ 1 $ row,
    $ c_1 $ has $ 1 $ row}

    % Step 5b: Move the double lines 

    \renewcommand{\moveboundaries}{
      $  \FlaTwoByTwo{A_{TL}}{A_{TR}}
      {A_{BL}}{A_{BR}}
      \leftarrow
        \FlaThreeByThreeBR{A_{00}}{*}{*}
      {a_{10}^T}{\alpha_{11}}{*}
      {A_{20}}{a_{21}}{A_{22}}
      $,
        $  \FlaTwoByOne{ B_T }
      { B_B }
      \leftarrow
        \FlaThreeByOneB{B_0}
      {b_1^T}
      {B_2}
      $
        ,
        $  \FlaTwoByOne{ C_T }
      { C_B }
      \leftarrow
        \FlaThreeByOneB{C_0}
      {c_1^T}
      {C_2}
      $
    }

% Step 6: State after repartitioning

}

% Step 7: State after moving of double lines

\renewcommand{\afterupdate}{

  {a_{21} b_1^T + A_{22} B_2 + C_2 = \widehat C_2}
}

  $
